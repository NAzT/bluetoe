478  template < typename T >
479  inline std::pair< std::uint8_t, T > deserialize( std::size_t write_size, const std::uint8_t* value )
480  {
481  static_assert( CHAR_BIT == 8, "needs porting!" );
482 
483  if ( write_size != sizeof( T ) )
484  return std::pair< std::uint8_t, T >{ error_codes::invalid_attribute_value_length, 0 };
485 
486  T result = 0;
487 
488  for ( const std::uint8_t* v = value + sizeof( T ); v != value; --v )
489  {
490  result = ( result << 8 ) | *( v - 1 );
491  }
492 
493  return std::pair< std::uint8_t, T >{ error_codes::success, result };
494  }
