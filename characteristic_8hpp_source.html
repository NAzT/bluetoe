244  static void fixup_auto_uuid( details::attribute_access_arguments& args )
245  {
246  // needs a 16 bit characteristic index!!!
247  std::uint16_t char_index = 1;
248 
249  static constexpr std::size_t uuid_offset = 3;
250 
251  int index_low = static_cast< int >( args.buffer_offset ) - static_cast< int >( uuid_offset );
252  int index_high = static_cast< int >( args.buffer_offset ) - static_cast< int >( uuid_offset + 1 );
253 
254  if ( index_low >= 0 && index_low < static_cast< int >( args.buffer_size ) )
255  args.buffer[ index_low ] ^= ( char_index & 0xff );
256 
257  if ( index_high >= 0 && index_high < static_cast< int >( args.buffer_size ) )
258  args.buffer[ index_high ] ^= ( char_index >> 8 );
259  }

261  /*
262  * the characteristic decalarion consists of 3 parts a Properties byte, two bytes index and 2 - 16 bytes UUID
263  */
264  static details::attribute_access_result char_declaration_access( details::attribute_access_arguments& args, std::uint16_t attribute_handle )
265  {
266  if ( args.type != details::attribute_access_type::read )
267  return details::attribute_access_result::write_not_permitted;
268 
269  const std::uint8_t properties[] = {
270  static_cast< std::uint8_t >(
271  ( value_type::has_read_access ? bits( details::gatt_characteristic_properties::read ) : 0 ) |
272  ( value_type::has_write_access ? bits( details::gatt_characteristic_properties::write ) : 0 ) |
273  ( value_type::has_write_without_response ? bits( details::gatt_characteristic_properties::write_without_response ) : 0 ) |
274  ( value_type::has_notification ? bits( details::gatt_characteristic_properties::notify ) : 0 ) |
275  ( value_type::has_indication ? bits( details::gatt_characteristic_properties::indicate ) : 0 ) )
276  };
277 
278  // the Characteristic Value Declaration must follow directly behind this attribute and has, thus the next handle
279  const std::uint8_t value_handle[] = {
280  static_cast< std::uint8_t >( ( attribute_handle +1 ) & 0xff ),
281  static_cast< std::uint8_t >( ( attribute_handle +1 ) >> 8 )
282  };
283 
284  static constexpr auto data_size = sizeof( properties ) + sizeof( value_handle ) + sizeof( uuid::bytes );
285 
286  if ( args.buffer_offset > data_size )
287  return details::attribute_access_result::invalid_offset;
288 
289  details::scattered_read_access( args.buffer_offset, properties, value_handle, uuid::bytes, args.buffer, args.buffer_size );
290 
291  if ( characteristic_or_service_uuid< Options... >::auto_generated_uuid )
292  fixup_auto_uuid( args );
293 
294  args.buffer_size = std::min< std::size_t >( data_size - args.buffer_offset, args.buffer_size );
295 
296  return details::attribute_access_result::success;
297  }
