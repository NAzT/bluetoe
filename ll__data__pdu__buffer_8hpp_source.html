400  template < std::size_t TransmitSize, std::size_t ReceiveSize, typename Radio >
401  write_buffer ll_data_pdu_buffer< TransmitSize, ReceiveSize, Radio >::next_transmit()
402  {
403  std::uint8_t* result = transmit_buffer_.next_end().buffer;
404 
405  if ( next_empty_ )
406  {
407  // if an empty buffer have to be resend, flag that there is more data
408  if ( result != nullptr )
409  result[ 0 ] |= more_data_flag;
410 
411  result = &empty_[ 0 ];
412  }
413  else if ( result == nullptr )
414  {
415  empty_[ 0 ] = ll_empty_id;
416  next_empty_ = true;
417  result = &empty_[ 0 ];
418 
419  // we created an PDU, so it have to have a new sequnce number
420  if ( sequence_number_ )
421  empty_[ 0 ] |= sn_flag;
422 
423  empty_sequence_number_ = sequence_number_;
424  sequence_number_ = !sequence_number_;
425  }
426  else
427  {
428  if ( transmit_buffer_.more_than_one() )
429  result[ 0 ] |= more_data_flag;
430  }
431 
432  // insert the next expected sequence for every attempt to send the PDU, because it could be that
433  // the slave is able to receive data, while the master is not able to.
434  result[ 0 ] = next_expected_sequence_number_
435  ? ( result[ 0 ] | nesn_flag )
436  : ( result[ 0 ] & ~nesn_flag );
437 
438  return write_buffer{ result, result[ 1 ] + ll_header_size };
439  }

441  template < std::size_t TransmitSize, std::size_t ReceiveSize, typename Radio >
442  void ll_data_pdu_buffer< TransmitSize, ReceiveSize, Radio >::acknowledge( bool nesn )
443  {
444  if ( next_empty_ )
445  {
446  if ( empty_sequence_number_ != nesn )
447  next_empty_ = false;
448  }
449  else
450  {
451  const read_buffer next = transmit_buffer_.next_end();
452 
453  // the transmit buffer could be empty if we receive without sending prior. That happens during testing
454  if ( next.empty() )
455  return;
456 
457  if ( static_cast< bool >( next.buffer[ 0 ] & sn_flag ) != nesn )
458  transmit_buffer_.pop_end( transmit_buffer() );
459  }
460  }
